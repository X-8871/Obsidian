```python
import torch
from torchvision.transforms import Lambda

print("="*60)
print(" 练习 1: 手搓狙击枪 (写一个 5 类的 One-Hot 转换器)")
print("="*60)
# 任务目标：
# 假设我们现在不是处理 10 种衣服了，而是处理一个只有 5 种类别的数据集。
# 请模仿我们在文档里学的写法，写一个专门处理这 5 类的 Lambda 转换器，并赋值给 my_target_transform
# 提示：原来是 torch.zeros(10...)，现在要改成多少？

# --- 请在下方写代码 ---
# my_target_transform = Lambda(lambda y: ...)
target_transform=Lambda(
    lambda y: torch.zeros(5, dtype=torch.float).scatter_(0, torch.tensor(y), value=1)
    )
# --- 代码结束 ---


print("\n" + "="*60)
print(" 练习 2: 狙击枪试射 (验证你的 One-Hot 编码)")
print("="*60)
# 任务目标：
# 假设我们当前抓到的一张图片的原始标签数字是 3。
# 请把数字 3 喂给刚才写好的 my_target_transform，并打印出结果，看看是否成功变成了独热灯泡！
# 预期输出应该类似于: tensor([0., 0., 0., 1., 0.])

# --- 请在下方写代码 ---
fake_label = 3
transformed_label =  target_transform(fake_label)

print(f"原始标签 {fake_label} 转换后变成了: {transformed_label}")


# --- 代码结束 ---


print("\n" + "="*60)
print(" 练习 3: 小挑战 (自定义图片数据转换)")
print("="*60)
# 任务目标：
# Lambda 不仅能处理标签(Y)，也能处理特征(X)。
# 假设我们拿到了一张图片张量 fake_image，它的像素值都在 0~1 之间。
# 请写一个 Lambda 转换器 (赋值给 my_image_transform)，它的作用是：把图片里所有的像素值都乘以 10！
# 提示：在 lambda 函数里，你可以直接对输入进行数学运算，比如 lambda x: x * 10

fake_image = torch.tensor([0.1, 0.5, 0.9]) # 假设这是一张极度简化的图片张量

# --- 请在下方写代码 ---
my_image_transform = lambda x: x * 10
result_image = my_image_transform(fake_image)
print(f"放大 10 倍后的图片像素是: {result_image}")


# --- 代码结束 ---

```

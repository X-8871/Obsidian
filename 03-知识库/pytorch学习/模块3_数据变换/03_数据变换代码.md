import torch
from torchvision.transforms import Lambda

print("="*60)
print(" 缁冧範 1: 鎵嬫悡鐙欏嚮鏋?(鍐欎竴涓?5 绫荤殑 One-Hot 杞崲鍣?")
print("="*60)
# 浠诲姟鐩爣锛?# 鍋囪鎴戜滑鐜板湪涓嶆槸澶勭悊 10 绉嶈。鏈嶄簡锛岃€屾槸澶勭悊涓€涓彧鏈?5 绉嶇被鍒殑鏁版嵁闆嗐€?# 璇锋ā浠挎垜浠湪鏂囨。閲屽鐨勫啓娉曪紝鍐欎竴涓笓闂ㄥ鐞嗚繖 5 绫荤殑 Lambda 杞崲鍣紝骞惰祴鍊肩粰 my_target_transform
# 鎻愮ず锛氬師鏉ユ槸 torch.zeros(10...)锛岀幇鍦ㄨ鏀规垚澶氬皯锛?
# --- 璇峰湪涓嬫柟鍐欎唬鐮?---
# my_target_transform = Lambda(lambda y: ...)
target_transform=Lambda(
    lambda y: torch.zeros(5, dtype=torch.float).scatter_(0, torch.tensor(y), value=1)
    )
# --- 浠ｇ爜缁撴潫 ---


print("\n" + "="*60)
print(" 缁冧範 2: 鐙欏嚮鏋瘯灏?(楠岃瘉浣犵殑 One-Hot 缂栫爜)")
print("="*60)
# 浠诲姟鐩爣锛?# 鍋囪鎴戜滑褰撳墠鎶撳埌鐨勪竴寮犲浘鐗囩殑鍘熷鏍囩鏁板瓧鏄?3銆?# 璇锋妸鏁板瓧 3 鍠傜粰鍒氭墠鍐欏ソ鐨?my_target_transform锛屽苟鎵撳嵃鍑虹粨鏋滐紝鐪嬬湅鏄惁鎴愬姛鍙樻垚浜嗙嫭鐑伅娉★紒
# 棰勬湡杈撳嚭搴旇绫讳技浜? tensor([0., 0., 0., 1., 0.])

# --- 璇峰湪涓嬫柟鍐欎唬鐮?---
fake_label = 3
transformed_label =  target_transform(fake_label)

print(f"鍘熷鏍囩 {fake_label} 杞崲鍚庡彉鎴愪簡: {transformed_label}")


# --- 浠ｇ爜缁撴潫 ---


print("\n" + "="*60)
print(" 缁冧範 3: 灏忔寫鎴?(鑷畾涔夊浘鐗囨暟鎹浆鎹?")
print("="*60)
# 浠诲姟鐩爣锛?# Lambda 涓嶄粎鑳藉鐞嗘爣绛?Y)锛屼篃鑳藉鐞嗙壒寰?X)銆?# 鍋囪鎴戜滑鎷垮埌浜嗕竴寮犲浘鐗囧紶閲?fake_image锛屽畠鐨勫儚绱犲€奸兘鍦?0~1 涔嬮棿銆?# 璇峰啓涓€涓?Lambda 杞崲鍣?(璧嬪€肩粰 my_image_transform)锛屽畠鐨勪綔鐢ㄦ槸锛氭妸鍥剧墖閲屾墍鏈夌殑鍍忕礌鍊奸兘涔樹互 10锛?# 鎻愮ず锛氬湪 lambda 鍑芥暟閲岋紝浣犲彲浠ョ洿鎺ュ杈撳叆杩涜鏁板杩愮畻锛屾瘮濡?lambda x: x * 10

fake_image = torch.tensor([0.1, 0.5, 0.9]) # 鍋囪杩欐槸涓€寮犳瀬搴︾畝鍖栫殑鍥剧墖寮犻噺

# --- 璇峰湪涓嬫柟鍐欎唬鐮?---
my_image_transform = lambda x: x * 10
result_image = my_image_transform(fake_image)
print(f"鏀惧ぇ 10 鍊嶅悗鐨勫浘鐗囧儚绱犳槸: {result_image}")

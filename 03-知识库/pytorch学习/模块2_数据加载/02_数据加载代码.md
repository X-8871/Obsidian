# --- 浠ｇ爜缁撴潫 ---

print("="*60)
# 浠诲姟鐩爣锛?# 鍋囪鎴戜滑鏈変竴涓秴绾х畝鍗曠殑鏁版嵁闆嗭紝閲岄潰鍙湁 100 寮犲叏榛戠殑鍥剧墖 (鍍忕礌鍏ㄦ槸 0)锛屾爣绛惧叏鏄?1銆?# 璇蜂綘琛ュ叏涓嬮潰杩欎釜鑷畾涔?Dataset 绫荤殑涓夊ぇ榄旀硶鍑芥暟銆?

        return self.total_samples

        image = torch.zeros(1, 28, 28)

import torch
from torch import nn
from torch.utils.data import DataLoader
from torchvision import datasets
from torchvision.transforms import ToTensor

# ==========================================
# 前置准备：这些我们前面几章都学过了，我直接帮你写好
# ==========================================
print("正在从网上进货采购图片 (FashionMNIST)...")
training_data = datasets.FashionMNIST(root="data", train=True, download=True, transform=ToTensor())
test_data = datasets.FashionMNIST(root="data", train=False, download=True, transform=ToTensor())

print("正在安排传菜员...")
train_dataloader = DataLoader(training_data, batch_size=64)
test_dataloader = DataLoader(test_data, batch_size=64)

class NeuralNetwork(nn.Module):
    def __init__(self):
        super().__init__()
        self.flatten = nn.Flatten()
        self.linear_relu_stack = nn.Sequential(
            nn.Linear(28*28, 512),
            nn.ReLU(),
            nn.Linear(512, 512),
            nn.ReLU(),
            nn.Linear(512, 10),
        )

    def forward(self, x):
        x = self.flatten(x)
        logits = self.linear_relu_stack(x)
        return logits

print("正在招募大厨...")
model = NeuralNetwork()

# 这是我们在文档里学过的测试营业厅代码，我直接提供给你
def test_loop(dataloader, model, loss_fn):
    size = len(dataloader.dataset)
    num_batches = len(dataloader)
    test_loss, correct = 0, 0

    with torch.no_grad():
        for X, y in dataloader:
            pred = model(X)
            test_loss += loss_fn(pred, y).item()
            correct += (pred.argmax(1) == y).type(torch.float).sum().item()

    test_loss /= num_batches
    correct /= size
    print(f"晚上营业战报: \n 好评准确率: {(100*correct):>0.1f}%, 平均 Loss: {test_loss:>8f} \n")


print("\n" + "="*60)
print(" 练习 1: 聘请两大主管 (定义品菜师和教练)")
print("="*60)
# 任务目标：
# 1. 聘请品菜师 loss_fn：使用 nn.CrossEntropyLoss()
# 2. 聘请主教练 optimizer：使用 torch.optim.SGD，把 model.parameters() 交给他，并设定学习率 lr=1e-3

# --- 请在下方写代码 ---
loss_fn = nn.CrossEntropyLoss()
optimizer = torch.optim.SGD(model.parameters(),lr=1e-3)

# --- 代码结束 ---


print("\n" + "="*60)
print(" 练习 2: 组装核心打工引擎 (完成 5 步曲)")
print("="*60)
# 任务目标：
# 补全下面 train_loop 里的 5 句核心口诀代码！

def train_loop(dataloader, model, loss_fn, optimizer):
    size = len(dataloader.dataset)
    for batch, (X, y) in enumerate(dataloader):
        
        # --- 请在下方写代码 (按顺序填入 5 步曲) ---
        
        # 1. 大厨做菜
        pred = model(X)
        
        # 2. 品菜师算分
        loss = loss_fn(pred,y)
        
        # 3. 教练撕毁旧追责单 (极其重要，用到 zero_grad)
        # ...
        optimizer.zero_grad()
        # 4. 追责系统算出新责任纸条 (反向传播，用到 backward)
        # ...
        loss.backward()
        # 5. 教练涂改大厨脑子 (用到 step)
        # ...
        optimizer.step()
        
        # --- 代码结束 ---

        # 打印进度条给老板看 (不用修改这里)
        if batch % 100 == 0:
            loss, current = loss.item(), batch * batch_size + len(X)
            print(f"进度: loss: {loss:>7f}  [{current:>5d}/{size:>5d}]")


print("\n" + "="*60)
print(" 练习 3: 发动引擎！(执行完整的 Epoch 循环)")
print("="*60)
# 任务目标：
# 我们要让大厨苦练 3 天 (epochs = 3)。
# 请写一个 for 循环 (遍历 epochs)。在每一天的循环里：
# 1. 打印今天是第几天
# 2. 调用 train_loop 函数，把 4 个天王全塞进去。
# 3. 调用 test_loop 函数，看看晚上的好评率是多少。

epochs = 10
batch_size = 64 # 给进度条用的变量

# --- 请在下方写代码 ---
# for t in range(epochs):
#     print(f"=== 第 {t+1} 天特训开始 ===")
#     ... (调用 train)
#     ... (调用 test)
for t in range(epochs):
    print(f"===第{t+1}天特训开始===")
    train_loop(train_dataloader,model,loss_fn,optimizer)
    test_loop(test_dataloader,model,loss_fn)
print("大厨特训完毕！")
# --- 代码结束 ---

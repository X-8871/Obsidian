import torch
from torch import nn
from torch.utils.data import DataLoader
from torchvision import datasets
from torchvision.transforms import ToTensor

# ==========================================
# 鍓嶇疆鍑嗗锛氳繖浜涙垜浠墠闈㈠嚑绔犻兘瀛﹁繃浜嗭紝鎴戠洿鎺ュ府浣犲啓濂?# ==========================================
print("姝ｅ湪浠庣綉涓婅繘璐ч噰璐浘鐗?(FashionMNIST)...")
training_data = datasets.FashionMNIST(root="data", train=True, download=True, transform=ToTensor())
test_data = datasets.FashionMNIST(root="data", train=False, download=True, transform=ToTensor())

print("姝ｅ湪瀹夋帓浼犺彍鍛?..")
train_dataloader = DataLoader(training_data, batch_size=64)
test_dataloader = DataLoader(test_data, batch_size=64)

class NeuralNetwork(nn.Module):
    def __init__(self):
        super().__init__()
        self.flatten = nn.Flatten()
        self.linear_relu_stack = nn.Sequential(
            nn.Linear(28*28, 512),
            nn.ReLU(),
            nn.Linear(512, 512),
            nn.ReLU(),
            nn.Linear(512, 10),
        )

    def forward(self, x):
        x = self.flatten(x)
        logits = self.linear_relu_stack(x)
        return logits

print("姝ｅ湪鎷涘嫙澶у帹...")
model = NeuralNetwork()

# 杩欐槸鎴戜滑鍦ㄦ枃妗ｉ噷瀛﹁繃鐨勬祴璇曡惀涓氬巺浠ｇ爜锛屾垜鐩存帴鎻愪緵缁欎綘
def test_loop(dataloader, model, loss_fn):
    size = len(dataloader.dataset)
    num_batches = len(dataloader)
    test_loss, correct = 0, 0

    with torch.no_grad():
        for X, y in dataloader:
            pred = model(X)
            test_loss += loss_fn(pred, y).item()
            correct += (pred.argmax(1) == y).type(torch.float).sum().item()

    test_loss /= num_batches
    correct /= size
    print(f"鏅氫笂钀ヤ笟鎴樻姤: \n 濂借瘎鍑嗙‘鐜? {(100*correct):>0.1f}%, 骞冲潎 Loss: {test_loss:>8f} \n")


print("\n" + "="*60)
print(" 缁冧範 1: 鑱樿涓ゅぇ涓荤 (瀹氫箟鍝佽彍甯堝拰鏁欑粌)")
print("="*60)
# 浠诲姟鐩爣锛?# 1. 鑱樿鍝佽彍甯?loss_fn锛氫娇鐢?nn.CrossEntropyLoss()
# 2. 鑱樿涓绘暀缁?optimizer锛氫娇鐢?torch.optim.SGD锛屾妸 model.parameters() 浜ょ粰浠栵紝骞惰瀹氬涔犵巼 lr=1e-3

# --- 璇峰湪涓嬫柟鍐欎唬鐮?---
loss_fn = nn.CrossEntropyLoss()
optimizer = torch.optim.SGD(model.parameters(),lr=1e-3)

# --- 浠ｇ爜缁撴潫 ---


print("\n" + "="*60)
print(" 缁冧範 2: 缁勮鏍稿績鎵撳伐寮曟搸 (瀹屾垚 5 姝ユ洸)")
print("="*60)
# 浠诲姟鐩爣锛?# 琛ュ叏涓嬮潰 train_loop 閲岀殑 5 鍙ユ牳蹇冨彛璇€浠ｇ爜锛?
def train_loop(dataloader, model, loss_fn, optimizer):
    size = len(dataloader.dataset)
    for batch, (X, y) in enumerate(dataloader):
        
        # --- 璇峰湪涓嬫柟鍐欎唬鐮?(鎸夐『搴忓～鍏?5 姝ユ洸) ---
        
        # 1. 澶у帹鍋氳彍
        pred = model(X)
        
        # 2. 鍝佽彍甯堢畻鍒?        loss = loss_fn(pred,y)
        
        # 3. 鏁欑粌鎾曟瘉鏃ц拷璐ｅ崟 (鏋佸叾閲嶈锛岀敤鍒?zero_grad)
        # ...
        optimizer.zero_grad()
        # 4. 杩借矗绯荤粺绠楀嚭鏂拌矗浠荤焊鏉?(鍙嶅悜浼犳挱锛岀敤鍒?backward)
        # ...
        loss.backward()
        # 5. 鏁欑粌娑傛敼澶у帹鑴戝瓙 (鐢ㄥ埌 step)
        # ...
        optimizer.step()
        
        # --- 浠ｇ爜缁撴潫 ---

        # 鎵撳嵃杩涘害鏉＄粰鑰佹澘鐪?(涓嶇敤淇敼杩欓噷)
        if batch % 100 == 0:
            loss, current = loss.item(), batch * batch_size + len(X)
            print(f"杩涘害: loss: {loss:>7f}  [{current:>5d}/{size:>5d}]")


print("\n" + "="*60)
print(" 缁冧範 3: 鍙戝姩寮曟搸锛?鎵ц瀹屾暣鐨?Epoch 寰幆)")
print("="*60)
# 浠诲姟鐩爣锛?# 鎴戜滑瑕佽澶у帹鑻︾粌 3 澶?(epochs = 3)銆?# 璇峰啓涓€涓?for 寰幆 (閬嶅巻 epochs)銆傚湪姣忎竴澶╃殑寰幆閲岋細
# 1. 鎵撳嵃浠婂ぉ鏄鍑犲ぉ
# 2. 璋冪敤 train_loop 鍑芥暟锛屾妸 4 涓ぉ鐜嬪叏濉炶繘鍘汇€?# 3. 璋冪敤 test_loop 鍑芥暟锛岀湅鐪嬫櫄涓婄殑濂借瘎鐜囨槸澶氬皯銆?
epochs = 10
batch_size = 64 # 缁欒繘搴︽潯鐢ㄧ殑鍙橀噺

# --- 璇峰湪涓嬫柟鍐欎唬鐮?---
# for t in range(epochs):
#     print(f"=== 绗?{t+1} 澶╃壒璁紑濮?===")
#     ... (璋冪敤 train)
#     ... (璋冪敤 test)
for t in range(epochs):
    print(f"===绗瑊t+1}澶╃壒璁紑濮?==")
    train_loop(train_dataloader,model,loss_fn,optimizer)
    test_loop(test_dataloader,model,loss_fn)
print("澶у帹鐗硅瀹屾瘯锛?)
